        {cherries.map((id) => (
          <Button
            key={id}
            variant="outline"
            size="lg"
            className={cn(
              "w-20 h-20 rounded-full flex items-center justify-center",
              caught.includes(id) && "bg-red-500 text-white",
              !caught.includes(id) && "bg-white text-red-500"

        ))}

  );
